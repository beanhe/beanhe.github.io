```
apt-get install -y php-apc
wget https://launchpad.net/ubuntu/+archive/primary/+files/icu_52.1.orig.tar.gz
tar xzf icu_52.1.orig.tar.gz
cd icu_52.1.orig/source
./configure
make && make install
pecl install intl
wget http://releases.wikimedia.org/mediawiki/1.25/mediawiki-1.25.2.tar.gz
tar xzf mediawiki-1.25.2.tar.gz -C /path/to/web/root
# Browse into  http://www.domain.com/mediawiki  to install
```
